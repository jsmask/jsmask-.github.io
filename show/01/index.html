// 敲小黑板，划重点啦~

function css3dTransform(ele,val) {

var arr = ["O", "ms", "Moz", "webkit", ""]

var len = arr.length;

for (var i = 0; i < len; i++) {

ele.style[arr[i] + "Transform"] = val;

}

}

// 再等等，马上啦~

var show3dBox = function() {

var box = document.querySelectorAll(".box");

var size = box.length;

var deg = 360 / size;

var dx = null,dy = null,sx = null,sy = null;

var loop = setInterval(function() {

if (dy == null) {

dy = 0,dx = 0;

}

dy++;

css3dTransform(document.body, "perspective(1000px) rotateY(" + dy + "deg)");

}, 30)

for (var i = 0; i < size; i++ ) {

css3dTransform(box[i], "rotateY(" + i * deg + "deg) translateZ(300px)");

}

}

// 见证奇迹的时刻！！！

show3dBox();
